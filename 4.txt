--1

SELECT E.ENAME, D.DNAME
FROM EMP E
JOIN DEPT D ON D.DEPTNO = E.DEPTNO
ORDER BY E.DEPTNO;

--2

SELECT E.ENAME, E.DEPTNO, D.DNAME
FROM EMP E
JOIN DEPT D ON D.DEPTNO = E.DEPTNO;

--3

SELECT E.ENAME, D.LOC, D.DNAME
FROM EMP E
JOIN DEPT D ON E.DEPTNO = D.DEPTNO AND E.SAL > 1500;

--4

SELECT E.ENAME, E.JOB, E.SAL, S.GRADE
FROM EMP E
JOIN SALGRADE S ON E.SAL BETWEEN S.LOSAL AND S.HISAL;

--5

SELECT E.ENAME, E.JOB, E.SAL, S.GRADE
FROM EMP E
JOIN SALGRADE S ON E.SAL BETWEEN S.LOSAL AND S.HISAL AND
S.GRADE = 3;

--6

SELECT E.ENAME, D.LOC
FROM EMP E
JOIN DEPT D ON E.DEPTNO = D.DEPTNO AND
UPPER(D.LOC) = 'DALLAS';

--7

SELECT E.ENAME, E.JOB, S.GRADE, D.DNAME
FROM EMP E
JOIN DEPT D ON E.DEPTNO = D.DEPTNO AND UPPER(E.JOB) != 'CLERK'
JOIN SALGRADE S ON E.SAL BETWEEN S.LOSAL AND S.HISAL;

--8
SELECT E.ENAME, E.JOB,12*E.SAL+NVL(COMM, 0) TOTAL_SAL,D.DEPTNO,D.DNAME,S.GRADEFROM EMP E

JOIN DEPT D ON E.DEPTNO = D.DEPTNO
JOIN SALGRADE S ON E.SAL BETWEEN S.LOSAL AND S.HISAL AND
(12*E.SAL+NVL(COMM, 0) = 36000 OR
UPPER(E.JOB) = 'CLERK');
